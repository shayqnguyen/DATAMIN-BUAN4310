# Normalization algorithm
train_norm <- train_df_rose
valid_norm <- valid_df

norm_values <- preProcess(train_df_rose[, -c(1)],
                          method = c("center", "scale"))

                                train_df_rose[, -c(1)])

# Apply to validation set

                                valid_df[, -c(1)])
